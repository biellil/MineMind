```typescript
// src/cognition/loop.smoke.test.ts
import { test, expect } from 'bun:test'
import { buildGraph } from './graph'
import { createControlState } from '../control/commands'
import { createSafetyState } from './safety'

// Mock minimo de Bot — fornece so o que buildWorldSnapshot le (ver snapshot.ts).
function makeMockBot(): any {
  const pos = { x: 0, y: 64, z: 0, distanceTo: (_o: any) => 0 }
  return {
    username: 'MineMind',
    health: 20, food: 20,
    entity: { position: pos },
    time: { timeOfDay: 1000 },
    entities: {},
    players: {},
    inventory: { items: () => [] },
    findBlocks: () => [],                 // mundo vazio -> arbitragem cai em 'exploring'
    blockAt: () => null,
    pathfinder: { goto: async () => {} }, // navigate resolve imediato
    on: () => {}, once: () => {},
  }
}

test('grafo roda 30 ticks via driver externo sem GraphRecursionError', async () => {
  const bot = makeMockBot()
  const graph = buildGraph({ bot, control: createControlState('autonomous'), safety: createSafetyState() })
  const cfg = { configurable: { thread_id: 'smoke' } }
  let last: any
  for (let i = 0; i < 30; i++) {
    last = await graph.invoke({}, cfg)    // se houvesse self-loop, estouraria aos 25 (Pitfall 1)
  }
  expect(last).toBeDefined()
})

test('memoria acumula eventos entre ticks (MemorySaver + thread_id)', async () => {
  const bot = makeMockBot()
  const graph = buildGraph({ bot, control: createControlState('autonomous'), safety: createSafetyState() })
  const cfg = { configurable: { thread_id: 'smoke-mem' } }
  let last: any
  for (let i = 0; i < 5; i++) last = await graph.invoke({}, cfg)
  expect(last.memory).toBeDefined()
  expect(Array.isArray(last.memory.events)).toBe(true)
  expect(last.memory.events.length).toBeGreaterThan(0)
})

test('modo paused mantem o agente em idle (sem skill)', async () => {
  const bot = makeMockBot()
  const control = createControlState('paused')
  const graph = buildGraph({ bot, control, safety: createSafetyState() })
  const last = await graph.invoke({}, { configurable: { thread_id: 'smoke-paused' } })
  expect(last.cogState).toBe('idle')
})
```
