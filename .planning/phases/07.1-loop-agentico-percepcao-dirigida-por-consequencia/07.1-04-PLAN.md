const skillAbort = new AbortController()

// Registrar listener ANTES de chamar a skill.
// hostileNearby é o gatilho de preempção v1 (D-18 — Policy: Phase 8 adicionará mais gatilhos).
const onHostileNearby = () => {
  log(`preemptando ${skill} — hostileNearby recebido`)
  skillAbort.abort('hostileNearby')
}
const { triggerBus } = deps
triggerBus.once('hostileNearby', onHostileNearby)

let emitOutcome: SkillOutcome | null = null
try {
  const params = skill === 'dig'
    ? { target, signal: skillAbort.signal }
    : { target: JSON.parse(target), signal: skillAbort.signal }

  // D-09 B: derivar memória do SkillResult OBSERVADO (Fase 7) — mantido como hoje
  const result = await skillRegistry[skill!]!(bot, params)
  const success = result.outcome === 'success'
  if (success) recordSuccess(safety)
  else recordFailure(safety, target, now())
  holder.lastObservedDelta = {
    skill,
    target,
    outcome: result.outcome,
    observed: result.observed,
    expected: result.expected,
    delta: { ...result.delta },
    at: now(),
  }
  holder.memory = push(holder.memory, {
    type: 'action',
    skill,
    target,
    outcome: result.outcome,
    observed: result.observed,
    expected: result.expected,
    result: success ? 'success' : 'failure',
    reason: result.reason,
    timestamp: now(),
  })
  log(`${result.outcome.toUpperCase()} ${skill} ${target} (${result.observed}/${result.expected})`)
  emitOutcome = result.outcome
} catch (err) {
  // Catch para exceções genuínas inesperadas (incluindo AbortError se skill não tratar o signal)
  const reason = err instanceof Error ? err.name : String(err)
  recordFailure(safety, target, now())
  holder.lastObservedDelta = { skill, target, outcome: 'error', observed: 0, expected: 0, delta: {}, at: now() }
  holder.memory = push(holder.memory, {
    type: 'action',
    skill,
    target,
    outcome: 'error',
    observed: 0,
    expected: 0,
    result: 'failure',
    reason,
    timestamp: now(),
  })
  log(`ERRO inesperado ${skill} ${target}: ${reason}`)
  emitOutcome = 'error'
} finally {
  // D-18/Pitfall 6: remover listener ANTES do abort() (evitar listener orphan)
  triggerBus.off('hostileNearby', onHostileNearby)
  skillAbort.abort()  // cleanup idempotente — no-op se já foi abortado

// Plan 03: actionFinished emitido APÓS grounding (Pitfall 1 da research)
// Quando Plan 03 estiver executado, esta linha já estará aqui.
