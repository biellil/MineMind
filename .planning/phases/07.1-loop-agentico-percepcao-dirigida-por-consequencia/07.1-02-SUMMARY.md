```typescript
export interface MechanicalStep {
  readonly skill: string   // ex.: 'dig'
  readonly target: string  // ex.: 'oak_log'
}

export function nextMechanicalStep(
  snapshot: WorldSnapshot,
  mode: ControlMode,
  safety: SafetyState,
```

**Lógica de classificação (ordem importa):**

| Gate | Condição | Retorno |
|------|----------|---------|
| Outcome (D-07) | `no_effect`/`partial` + `shouldAbandon` ativo | `null` → LLM |
| Modo paused | `mode === 'paused'` | `null` |
| Socializing | `hasNearbyPlayer` | `null` → LLM (decisão real) |
| Standby | `mode === 'standby'` | `null` |
| Gathering | `highestPriorityGatherTarget` com `cooledDownTargets` | `{ skill: 'dig', target }` |
| Exploring | fallback | `null` → LLM (posição decidida pelo LLM) |

**`arbitrate()` preservada intacta** — usada pelos nós `analyze`/`nodes.ts` sem alteração.

### Task 2 — `pickTrigger` em `src/cognition/loop.ts`

```typescript
function pickTrigger(holder: CognitiveStateHolder): DeliberationTrigger {
  // D-07: outcome do SkillResult como gatilho primário
  const lastOutcome = holder.lastObservedDelta?.outcome
  if (lastOutcome === 'no_effect' || lastOutcome === 'partial') {
    return 'need_threshold'  // força re-deliberação mais cedo que 'periodic'
  }
  // D-08: need threshold
  const crossed = holder.needs.some((n) => urgency(n, t, mcfg) >= mcfg.goalThreshold)
  return crossed ? 'need_threshold' : 'periodic'
}
```

`void deliberator.maybeDeliberate(...)` preservado — nenhuma chamada `await` introduzida (D-08 single-flight).

## Integração com o Loop Existente

```
SkillResult.outcome (Fase 7)
       │
```
